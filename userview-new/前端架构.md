# 前端项目结构 
```
src/
├── assets/                # 静态资源目录（图片、样式等）
│   └── logo.png
├── components/            # 公共组件目录
│   └── HelloWorld.vue     # 示例组件
├── router/                # 路由配置目录
│   └── index.js
├── service/               # 服务层目录（API调用等）
├── store/                 # Vuex状态管理目录
│   └── index.js
├── views/                 # 页面视图目录
│   ├── HomeView.vue       # 首页
│   ├── LoginView.vue      # 登录页面
│   ├── RegisterView.vue   # 注册页面
│   ├── MainView.vue       # 主页面
│   ├── user/              # 用户相关页面
│   │   ├── DishesList.vue       # 菜品列表
│   │   ├── CanteensList.vue     # 食堂列表
│   │   ├── BlogsList.vue        # 博客列表
│   │   ├── AddBlog.vue          # 新增博客
│   │   ├── CanteenDetails.vue   # 食堂详情
│   │   ├── BlogDetails.vue      # 博客详情
│   │   └── UserProfile.vue      # 用户信息
│   ├── boss/              # 管理员相关页面
│   │   ├── StatisticsAnalytics.vue # 统计分析
│   │   ├── RoleManage.vue         # 角色管理
│   │   └── ProgrameManage.vue     # 项目管理
│   └── admin/             # 超级管理员相关页面
│       ├── BulletinsManage.vue   # 公告管理
│       ├── HomepageManage.vue    # 首页管理
│       ├── DataManage.vue        # 数据管理
│       └── BlogsManage.vue       # 博客管理
├── App.vue                # 根组件
└── main.js                # 应用入口文件
```
